# EK2 + RM compatch aging tuning
# Overrides RM values with EK2 lifespan-aware pacing.

minimum_height_growth = {
	value = age
	if = {
		limit = {
			is_alive = yes
			has_variable = height_genetic
		}
		add = {
			value = 50
			subtract = var:height_genetic
			divide = 2
			min = 0
		}
	}

	# Baseline human-ish growth floor.
	divide = 25

	# Long-lived species reach adult proportions later.
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_2 } }
		divide = 1.2
	}
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_3 } }
		divide = 1.45
	}
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_6 } }
		divide = 2.2
	}
	max = 1
}

maximum_height_growth = {
	value = 0.1
	multiply = age

	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_2 } }
		multiply = 0.9
	}
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_3 } }
		multiply = 0.75
	}
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_6 } }
		multiply = 0.5
	}
}

puberty_aging_value = {
	value = 0
	add = {
		value = age
		divide = 10
		multiply = 0.4
	}
	add = {
		if = {
			limit = { has_variable = puberty_value }
			value = var:puberty_value
			divide = 1.25
			multiply = 0.6
		}
		else = {
			value = age
			subtract = 10
			divide = 15
			multiply = 0.6
		}
	}

	# Slow apparent aging for long-lived heritages.
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_2 } }
		multiply = 0.85
	}
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_3 } }
		multiply = 0.65
	}
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_6 } }
		multiply = 0.35
	}

	max = 1
}
