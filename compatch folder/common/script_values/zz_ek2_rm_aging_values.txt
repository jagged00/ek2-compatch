# EK2 + RM compatch aging tuning
# Overrides RM values with EK2 lifespan-aware pacing.

minimum_height_growth = {
	value = age
	if = {
		limit = {
			is_alive = yes
			has_variable = height_genetic
		}
		add = {
			value = 50
			subtract = var:height_genetic
			divide = 2
			min = 0
		}
	}

	# Keep RM baseline growth floor so children do not stall undersized.
	divide = 25
	max = 1
}

maximum_height_growth = {
	value = 0.1
	multiply = age
}

puberty_aging_value = {
	value = 0
	add = {
		if = {
			limit = { has_variable = puberty_start_age }
			value = age
			divide = var:puberty_start_age
			multiply = 0.4
		}
		else = {
			value = age
			divide = 10
			multiply = 0.4
		}
	}
	if = {
		limit = { has_variable = puberty_value }
		add = {
			value = var:puberty_value
			divide = 1.25
			multiply = 0.6
		}
	}
	else = {
		add = {
			value = age
			subtract = 10
			divide = 15
			multiply = 0.6
		}
	}

	# Slow apparent aging for long-lived heritages.
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_2 } }
		multiply = 0.85
	}
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_3 } }
		multiply = 0.65
	}
	if = {
		limit = { culture = { has_cultural_parameter = heritage_lifespan_6 } }
		multiply = 0.35
	}

	max = 1
}


ek2_rm_visual_age_base_value = {
	if = {
		limit = {
			ek_human_age_equivalent >= 35
			has_trait = beauty_good
		}
		value = ek_human_age_equivalent
		subtract = 35
		multiply = 0.02
		min = 0
		max = 1.5
	}
	else_if = {
		limit = { ek_human_age_equivalent >= 25 }
		value = ek_human_age_equivalent
		subtract = 25
		multiply = 0.02
		min = 0
		max = 1.5
	}
	else = {
		value = 0
	}
}
