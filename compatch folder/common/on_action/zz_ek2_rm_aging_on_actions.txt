# EK2 + RM visual aging compatibility overrides
# Overrides RM visual age on_actions to apply EK2 lifespan pacing.

on_add_visual_age_variable = {
	trigger = {
		is_alive = yes
		Not = { is_immortal = yes }
		trigger_if = {
			limit = { is_alive = yes }
			not = { has_variable = visual_age }
			Not = { has_character_flag = is_newborn }
		}
	}
	effect = {
		set_variable = {
			name = visual_age
			value = {
				if = {
					limit = {
						age >= 35
						has_trait = beauty_good
					}
					value = age
					subtract = 35
					multiply = 0.02
					min = 0
					max = 1.5
				}
				else_if = {
					limit = { age >= 25 }
					value = age
					subtract = 25
					multiply = 0.02
					min = 0
					max = 1.5
				}
				else = {
					value = 0
				}

				# EK2 lifespan pacing.
				if = {
					limit = { culture = { has_cultural_parameter = heritage_lifespan_2 } }
					multiply = 0.85
				}
				if = {
					limit = { culture = { has_cultural_parameter = heritage_lifespan_3 } }
					multiply = 0.65
				}
				if = {
					limit = { culture = { has_cultural_parameter = heritage_lifespan_6 } }
					multiply = 0.35
				}
			}
		}
	}
}

on_increase_visual_age_variable = {
	trigger = {
		is_alive = yes
		Not = { is_immortal = yes }
		trigger_if = {
			limit = { is_alive = yes }
			has_variable = visual_age
			Not = { has_character_flag = is_newborn }
		}
		trigger_if = {
			limit = { has_trait = beauty_good }
			age > 35
		}
		trigger_else = {
			age > 25
		}
	}
	effect = {
		change_variable = {
			name = visual_age
			add = {
				value = 0.02
				if = {
					limit = { culture = { has_cultural_parameter = heritage_lifespan_2 } }
					multiply = 0.85
				}
				if = {
					limit = { culture = { has_cultural_parameter = heritage_lifespan_3 } }
					multiply = 0.65
				}
				if = {
					limit = { culture = { has_cultural_parameter = heritage_lifespan_6 } }
					multiply = 0.35
				}
			}
		}
		if = {
			limit = { var:visual_age > 1.5 }
			remove_variable = visual_age
			set_variable = {
				name = visual_age
				value = 1.5
			}
		}
		if = {
			limit = {
				has_trait = infirm
				var:visual_age < 0.5
			}
			remove_trait = infirm
		}
	}
}
